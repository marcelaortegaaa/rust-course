1//! This library contains utils for normalizing inputs
2//! The function available turns Strings into tuples that separate value and unit.
3//! # Example
4//! ```
5//! use library_project::prepare_input;
6//! let input = "3kg".to_string();
7//! let normalized_input = prepare_input(&input);
8//! ```
9
10pub mod byte_unit;
11
12/// This function takes a String and normalizes it into a tuple with an unsigned integer and a lowercase String.
13/// It's meant to be used on strings that contain a value and a unit.
14/// It doesn't matter if your value and unit are separate or together
15/// # Example
16/// ```
17/// use library_project::prepare_input;
18/// let input = "3kg".to_string();
19/// let normalized_input = prepare_input(&input);
20/// ```
21pub fn prepare_input(s: &String) -> (u64, String) {
22    let reduced = s.to_lowercase().split_whitespace().collect::<String>();
23
24    match reduced.find(|c: char| c.is_alphabetic()) {
25        Some(i) => {
26            let (left, right) = reduced.split_at(i);
27            (left.parse().unwrap(), right.to_string())
28        }
29        None => (reduced.parse().unwrap(), "".to_string()),
30    }
31}
32
33/// Returns errors when turning inputs into bytes
34#[derive(Debug)]
35pub enum ParseSizeError {
36    UnknownUnit(String),
37    Overflow,
38}
39
40/// Turns value and unit into bytes
41/// ```
42/// use library_project::to_base_bytes;
43/// to_base_bytes(45, "mb");
44/// ```
45pub fn to_base_bytes(number: u64, unit_str: &str) -> Result<u64, ParseSizeError> {
46    let unit = byte_unit::ByteUnit::from_str(unit_str)
47        .ok_or_else(|| ParseSizeError::UnknownUnit(unit_str.to_string()))?;
48
49    number
50        .checked_mul(unit.multiplier())
51        .ok_or(ParseSizeError::Overflow)
52}
