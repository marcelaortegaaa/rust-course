4pub fn prepare_input(s: String) -> (u64, String) {
5    let reduced = s.to_lowercase().split_whitespace().collect::<String>();
6
7    match reduced.find(|c: char| c.is_alphabetic()) {
8        Some(i) => {
9            let (left, right) = reduced.split_at(i);
10            (left.parse().unwrap(), right.to_string())
11        }
12        None => (reduced.parse().unwrap(), "".to_string()),
13    }
14}

16#[derive(Debug, Clone, Copy)]
17enum Unit {
18    Bytes,
19    Kilobytes,
20    Megabytes,
21    Gigabytes,
22}
23
24impl Unit {
25    fn from_str(s: &str) -> Option<Self> {
26        match s {
27            "b" | "byte" | "bytes" => Some(Unit::Bytes),
28            "kb" | "kbs" | "kilobyte" | "kilobytes" => Some(Unit::Kilobytes),
29            "mb" | "megabyte" | "megabytes" => Some(Unit::Megabytes),
30            "gb" | "gigabyte" | "gigabytes" => Some(Unit::Gigabytes),
31            "" => Some(Unit::Bytes),
32            _ => None,
33        }
34    }
35
36    fn multiplier(self) -> u64 {
37        match self {
38            Unit::Bytes => 1,
39            Unit::Kilobytes => 1_000,
40            Unit::Megabytes => 1_000_000,
41            Unit::Gigabytes => 1_000_000_000,
42        }
43    }
44
45    fn label(self) -> &'static str {
46        match self {
47            Unit::Bytes => "bytes",
48            Unit::Kilobytes => "kilobytes",
49            Unit::Megabytes => "megabytes",
50            Unit::Gigabytes => "gigabytes",
51        }
52    }
53}

55#[derive(Debug)]
56enum ParseSizeError {
57    UnknownUnit(String),
58    Overflow,
59}
60
61pub fn to_base_bytes(number: u64, unit_str: &str) -> Result<u64, ParseSizeError> {
62    let unit = Unit::from_str(unit_str)
63        .ok_or_else(|| ParseSizeError::UnknownUnit(unit_str.to_string()))?;
64
65    number
66        .checked_mul(unit.multiplier())
67        .ok_or(ParseSizeError::Overflow)
68}
